This rule detects invalid method comment

Noncompliant Code Example

	public boolean isNameEmpty(String name) {
	  return name.length() == 0; //name could be null
	}

	public boolean isNameEmpty() {
		String name = null;//assign null value
		return name.length() == 0; // name: Please avoid NullPointerException
	}

	public boolean isNameEmpty(){
		String name;//don't assign value
		return name.length() == 0; // name: Please avoid NullPointerException
	}

Compliant Solution

	public boolean isNameEmpty() {
		String name = "Your name";
		return name.length() == 0;
	}

	public boolean isNameEmpty() throws NullPointerException {
		String name = null;
		return name.length() == 0;
	}

	public boolean isNameEmpty(){
		String name = null;
		if(name==null) {
			//TO DO
		}
		return name.length() == 0;
	}

	public boolean isNameEmpty() {
		String name = null;
		try {
			return name.length() == 0;
		} catch (NullPointerException e) {
			Log.error(e.getMessage());
			return 0;
		}
	}

Note: You can throw or catch an Exception instead NullPointerException.
